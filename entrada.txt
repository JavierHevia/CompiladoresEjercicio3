Function Void tablas(Number param1, Number param2){

	a=0;

		a=a+1;
		b=0;

			b=b+1;
			Print(a & " * " & b & " = " & a * b);
		}
		Print("----------------");
	}
}

Function Number factorial(Number n){
    if (n==0){
        return 1;
	}
    else{
        return n*(factorial(n-1));
	}
}

Function Number factorial2(Number n){
    Declare Number fact;
	fact=1;
    Declare Number i;
    if (n==0){
        fact=1;
	}
    else{
		For(i=1;i<n;i=i+1){
			fact=fact*i;
		}
	}
 return fact;
}
